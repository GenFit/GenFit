# Create a library called "libgenfit" which includes the source files given in
# the array .
# The extension is already found.  Any number of sources could be listed here.

set(INCLUDE_DIRECTORIES
${ROOT_INCLUDE_DIR} 
${CLHEP_INCLUDE_DIR}
${CMAKE_SOURCE_DIR}/genfit
${CMAKE_SOURCE_DIR}/geane
${CMAKE_SOURCE_DIR}/GenfitTools/trackrep/GeaneTrackRep2
${GEANT3_INCLUDE_DIR}
)

include_directories( ${INCLUDE_DIRECTORIES})

set(LINK_DIRECTORIES
${ROOT_LIBRARY_DIR}
)

link_directories( ${LINK_DIRECTORIES})

set(GENFIT_GEANE_SRCS GeaneTrackRep2.cxx GeaneMCApplication.cxx)

if(RULE_CHECKER_FOUND)
  CHECK_RULES("${GENFIT_GEANE_SRCS}" "${INCLUDE_DIRECTORIES}" GENFIT_GEANE_RULES)
endif(RULE_CHECKER_FOUND)

# fill list of header files from list of source files
# by exchanging the file extension
CHANGE_FILE_EXTENSION(*.cxx *.h GENFIT_GEANE_HEADERS "${GENFIT_GEANE_SRCS}")

set(GENFIT_GEANE_LINKDEF  genfitGeaneLinkDef.h)
set(GENFIT_GEANE_DICTIONARY ${CMAKE_CURRENT_BINARY_DIR}/genfitGeaneDict.cxx) 

ROOT_GENERATE_DICTIONARY("${GENFIT_GEANE_HEADERS}" "${GENFIT_GEANE_LINKDEF}" "${GENFIT_GEANE_DICTIONARY}" "${INCLUDE_DIRECTORIES}")

SET(GENFIT_GEANE_SRCS ${GENFIT_GEANE_SRCS} ${GENFIT_GEANE_DICTIONARY})

add_library(genfitGeane SHARED ${GENFIT_GEANE_SRCS})
target_link_libraries(genfitGeane ${ROOT_LIBRARIES})
set_target_properties(genfitGeane PROPERTIES ${FAIRROOT_LIBRARY_PROPERTIES})
################ install ###################
install(TARGETS genfitGeane DESTINATION ${CMAKE_BINARY_DIR}/lib)
